prontera,166,200,5	script	Reset Girl::resetGirl	124,{
	set .@stzeny,BaseLevel*10000;
	set .@skzeny,JobLevel*10000;
	set .@bzeny,((.@stzeny*9)+(.@skzeny*9))/10;
	mes "[Reset Girl]";
	mes "I am the Reset Girl.";
	mes "Reset Stats: "+.@stzeny+"z";
	mes "Reset Skills: "+.@skzeny+"z";
	mes "Reset Both: "+.@bzeny+"z";
	mes "Platinum Coins: 5 per reset, 10 for both";
	mes "Please select the service you want:";
	next;

	switch(select("^FF3355Reset Skills:Reset Stats:Reset Both^000000:Free Reset?:Platinum Coins:Quit")) {
		case 1:
			mes "[Reset Girl]";
			if (Zeny < .@skzeny) {
				mes "Sorry, you don't have enough Zeny.";
				close;
			}
			mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
			set Zeny,zeny-.@skzeny;
			ResetSkill;
			close;
		case 2:
			mes "[Reset Girl]";
			if (Zeny < .@stzeny) {
				mes "Sorry, you don't have enough Zeny.";
				close;
			}
			mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
			set Zeny,zeny-.@stzeny;
			ResetStatus;
			close;
		case 3:
			mes "[Reset Girl]";
			if (Zeny < .@bzeny) {
				mes "Sorry, you don't have enough Zeny.";
				close;
			}
			mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
			set Zeny,zeny-.@bzeny;
			ResetSkill;
			ResetStatus;
			close;
		case 4:
			mes "[Reset Girl]";
			mes "Oh dear! Were you part of that deal?";
			mes "Just let me check my paperwork here...";
			mes "...";
			next;
			if (redro_reset == 1) {
				mes "[Reset Girl]";
				mes "I'm sorry, but my records indicate";
				mes "that you've already used your reset!";
				close;
			}
				mes "[Reset Girl]";
				mes "Oh! There you are!";
				mes "We can get started right away!";
				next;
				
				mes "[Reset Girl]";
				mes "Which service can I interest you in?";
			switch(select("^FF3355Reset Skills:Reset Stats:Reset Both^000000:Quit")){
			case 1:
				mes "[Reset Girl]";
				mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
				ResetSkill;
				set redro_reset,1;
				close;
			case 2:
				mes "[Reset Girl]";
				mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
				ResetStatus;
				set redro_reset,1;
				close;
			case 3:
				mes "[Reset Girl]";
				mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
				ResetSkill;
				ResetStatus;
				set redro_reset,1;
				close;
			case 4:
				close;
				}
		
		case 5:
		 	mes "[Reset Girl]";
		 	mes "Platinum Coins are shiny!";
		 	mes "Reset Stats: 5 coins";
			mes "Reset Skills: 5 coins";
			mes "Reset Both: 10 coins";
		 	switch(select("^FF3355Reset Skills:Reset Stats:Reset Both^000000:Quit")){
			case 1:
				if (countitem(677) >= 5) {
					mes "[Reset Girl]";
					mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
					delitem 677,5;
					ResetSkill;
					close;
				} else {
					mes "[Reset Girl]";
					mes "You need 5 coins to reset skills";
					close;
				}
		 	case 2:
				if (countitem(677) >= 5) {
					mes "[Reset Girl]";
					mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
					delitem 677,5;
					ResetStatus;
					close;
				} else {
					mes "[Reset Girl]";
					mes "You need 5 coins to reset stats";
					close;
				}
		 	case 3:
				if (countitem(677) >= 10) {
					mes "[Reset Girl]";
					mes "Alright, here we go now.. Remember, some changes might not take effect until you log back on!";
					delitem 677,10;
					ResetSkill;
					ResetStatus;
					close;
				} else {
					mes "[Reset Girl]";
					mes "You need 10 coins for a full reset";
					close;
				}
		 	case 4:
		 		close;
		 	}

		case 6:
			close;
	}
}
